Replace it with this:

suspend fun imageUrl(
    path: String,
    width: Int = 300,
    height: Int = 450
): String = withContext(Dispatchers.IO) {
    val base = serverBase()
    val token = savedToken() ?: error("Not linked")

    if (path.isBlank()) {
        ""
    } else {
        val cleanBase = base.trimEnd('/')
        val safePath = encode(path)
        val safeToken = encode(token)

        "$cleanBase/photo/:/transcode" +
            "?width=$width" +
            "&height=$height" +
            "&minSize=1" +
            "&upscale=1" +
            "&url=$safePath" +
            "&X-Plex-Token=$safeToken"
    }
}
